    数据集：学生成绩预测

    我们用一份 400 名学生的合成数据集。每行代表一名学生，有四个输入特征和一个结果变量。我们会同时预测连续分数（回归）和通过/不通过标签（分类）。

    列名类型说明
    study_hoursfloat每周学习时长（0–20 小时），含部分异常值
    prev_scorefloat上次考试分数（0–100），有少量缺失值
    attendancefloat出勤率（0.0–1.0）
    sleep_hoursfloat平均每晚睡眠时长（3–10 小时）
    support_levelstr辅导支持等级："none"、"basic"、"intensive"
    final_scorefloat目标变量：期末考试分数（0–100）

    Step 0 — 生成合成数据集

    我们用 NumPy 在代码里生成数据，这样整个项目完全自给自足，不依赖外部文件。数据里内置了真实的噪声、缺失值和异常值。

    Python — generate_data.py（运行一次）
    import numpy as np
import pandas as pd

rng = np.random.default_rng(seed=2024)
n = 400

study_hours   = rng.uniform(0, 16, n)
prev_score    = rng.normal(65, 18, n).clip(0, 100)
attendance    = rng.beta(7, 2, n)           # 偏向高出勤率
sleep_hours   = rng.normal(7, 1.2, n).clip(3, 10)
support_level = rng.choice(["none", "basic", "intensive"],
                             p=[0.5, 0.3, 0.2], size=n)

# 真实的分数生成逻辑（加噪声）
support_bonus = {"none": 0, "basic": 5, "intensive": 12}
bonus = np.array([support_bonus[s] for s in support_level])

final_score = (
    0.35 * study_hours * 4           # 学习时长 → 最多约 22 分
  + 0.40 * prev_score                # 历史成绩权重最高
  + 0.15 * attendance * 100          # 出勤率 → 最多 15 分
  + 0.10 * (sleep_hours - 4) * 5    # 睡眠 → 小幅加成
  + bonus
  + rng.normal(0, 8, n)             # 噪声
).clip(0, 100)

df = pd.DataFrame({
    "study_hours":   study_hours,
    "prev_score":    prev_score,
    "attendance":    attendance,
    "sleep_hours":   sleep_hours,
    "support_level": support_level,
    "final_score":   final_score,
})

# 注入真实的"脏数据"问题
df.loc[rng.choice(n, 20, replace=False), "prev_score"] = np.nan   # 缺失值
df.loc[rng.choice(n, 5,  replace=False), "study_hours"] *= 3      # 异常值
df.loc[rng.choice(n, 8,  replace=False), "attendance"] = rng.uniform(0, 0.1, 8)

df = df.round(2)
df.to_csv("students.csv", index=False)
print(f"已保存 {len(df)} 行到 students.csv")

    Step 2 — 数据清洗

    Python — 02_clean.py
    import pandas as pd
import numpy as np

df = pd.read_csv("students.csv")

def clean(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()

    # ── 缺失值处理 ──
    # 用中位数填充 prev_score（中位数对异常值更鲁棒）
    df["prev_score"] = df["prev_score"].fillna(df["prev_score"].median())

    # ── 截断 study_hours 的异常值（上限 20 小时/周）──
    df["study_hours"] = df["study_hours"].clip(upper=20.0)

    # ── 修正不可能的出勤率值 ──
    df["attendance"] = df["attendance"].clip(0.0, 1.0)

    # ── 确保类型正确 ──
    df["support_level"] = df["support_level"].str.strip().str.lower()

    return df

df_clean = clean(df)

# 验证
print("清洗后缺失值：", df_clean.isnull().sum().sum())
print("study_hours 最大值：", df_clean["study_hours"].max())
print("attendance 范围：", df_clean["attendance"].min(), "–", df_clean["attendance"].max())

df_clean.to_csv("students_clean.csv", index=False)
print(f"\n已保存清洗后数据。形状：{df_clean.shape}")

    Step 3 — 特征工程

    Python — 03_features.py
    import pandas as pd
import numpy as np

df = pd.read_csv("students_clean.csv")

def engineer_features(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()

    # ── 有序类别编码 ──
    support_map = {"none": 0, "basic": 1, "intensive": 2}
    df["support_num"] = df["support_level"].map(support_map)

    # ── 交互特征 ──
    # 学习效率 = 学习时长 × 出勤率
    df["study_efficiency"] = df["study_hours"] * df["attendance"]

    # 睡眠质量分——睡太少和睡太多都扣分
    df["sleep_quality"] = 1.0 - (np.abs(df["sleep_hours"] - 7.5) / 4.5).clip(0, 1)

    # ── 分类任务的二值目标 ──
    df["passed"] = (df["final_score"] >= 60).astype(int)

    return df

df_feat = engineer_features(df)

# 检查结果
print(df_feat[["study_efficiency", "sleep_quality", "support_num", "passed"]].describe().round(2))
df_feat.to_csv("students_features.csv", index=False)
print(f"\n特征数据已保存。列名：{list(df_feat.columns)}")

    Step 4 — 建模

    Python — 04_model.py
    import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LinearRegression, LogisticRegression
from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import (
    r2_score, mean_absolute_error,
    accuracy_score, classification_report
)

df = pd.read_csv("students_features.csv")

FEATURES = ["study_hours", "prev_score", "attendance",
            "sleep_hours", "support_num", "study_efficiency", "sleep_quality"]

X = df[FEATURES]
y_reg  = df["final_score"]    # 回归目标（连续值）
y_clf  = df["passed"]          # 分类目标（二值）

X_train, X_test, yr_train, yr_test, yc_train, yc_test = train_test_split(
    X, y_reg, y_clf, test_size=0.2, random_state=42, stratify=y_clf
)

print(f"训练集：{len(X_train)} 条 | 测试集：{len(X_test)} 条")
print(f"通过率（训练）：{yc_train.mean():.1%}  （测试）：{yc_test.mean():.1%}\n")

# ────────────────────────────────
#  回归——预测 final_score
# ────────────────────────────────
reg_pipe = Pipeline([
    ("scaler", StandardScaler()),
    ("model",  LinearRegression()),
])
reg_pipe.fit(X_train, yr_train)
yr_pred = reg_pipe.predict(X_test)

print("=== 线性回归 ===")
print(f"R²:  {r2_score(yr_test, yr_pred):.3f}")
print(f"MAE: {mean_absolute_error(yr_test, yr_pred):.2f} 分\n")

# 5 折交叉验证 R²
cv_r2 = cross_val_score(reg_pipe, X, y_reg, cv=5, scoring="r2")
print(f"5 折 CV R²: {cv_r2.mean():.3f} ± {cv_r2.std():.3f}\n")

# ────────────────────────────────
#  分类——预测 passed
# ────────────────────────────────
log_pipe = Pipeline([
    ("scaler", StandardScaler()),
    ("model",  LogisticRegression(max_iter=1000)),
])
log_pipe.fit(X_train, yc_train)
yc_pred = log_pipe.predict(X_test)

print("=== 逻辑回归 ===")
print(f"准确率: {accuracy_score(yc_test, yc_pred):.3f}")
print(classification_report(yc_test, yc_pred, target_names=["不通过", "通过"]))

# 用决策树做对比
tree_pipe = Pipeline([
    ("model", DecisionTreeClassifier(max_depth=5, min_samples_leaf=8, random_state=42))
])
tree_pipe.fit(X_train, yc_train)
print("=== 决策树 ===")
print(f"准确率: {tree_pipe.score(X_test, yc_test):.3f}")

    解读结果

    有了数字之后，工作还没结束——你需要问自己：这些结果合理吗？它们意味着什么？

    读懂系数
    因为我们在拟合前对特征做了标准化，线性回归的系数可以直接比较大小。系数绝对值越大，说明该特征对预测分数的影响越强。如果 prev_score 的系数最大，意味着历史成绩是最强的预测信号——这符合直觉，也给了我们信心：模型学到的是真实规律，而不是噪声。

    # 对两个假设学生做预测，验证模型输出是否合理
hypotheticals = pd.DataFrame([

    管道思维 (Pipeline Thinking)

    注意我们的做法：每一步都是一个纯函数（第 4 章），接收一个 DataFrame，返回一个新 DataFrame。整个 pipeline 就是函数的组合链：

    Python — 用一个函数跑完整个 pipeline
    def run_pipeline(raw_csv: str) -> dict:
    """
    完整的数据科学流水线——从加载到评估一键跑完。
    每个步骤都可以独立测试和替换。
    """
    # 加载
    df = pd.read_csv(raw_csv)

    # 清洗
    df = clean(df)                # Step 2

    # 特征工程
    df = engineer_features(df)    # Step 3

    # 切分
    X = df[FEATURES]
    y_reg = df["final_score"]
    y_clf = df["passed"]
    X_train, X_test, yr_train, yr_test, yc_train, yc_test = train_test_split(
        X, y_reg, y_clf, test_size=0.2, random_state=42, stratify=y_clf
    )

    # 训练
    reg_pipe = Pipeline([("scaler", StandardScaler()), ("model", LinearRegression())])
    clf_pipe = Pipeline([("scaler", StandardScaler()), ("model", LogisticRegression(max_iter=1000))])
    reg_pipe.fit(X_train, yr_train)
    clf_pipe.fit(X_train, yc_train)

    # 评估
    return {
        "regression_r2":       r2_score(yr_test, reg_pipe.predict(X_test)),
        "regression_mae":      mean_absolute_error(yr_test, reg_pipe.predict(X_test)),
        "classification_acc":  clf_pipe.score(X_test, yc_test),
        "n_train": len(X_train),
        "n_test":  len(X_test),
    }

results = run_pipeline("students.csv")
for k, v in results.items():
    print(f"{k:25s}: {v:.3f}" if isinstance(v, float) else f"{k:25s}: {v}")
    你刚刚构建了什么

    这是一个真实的、可用于生产的数据科学工作流结构。clean()、engineer_features() 和最终的 pipeline 都可以独立测试、版本控制和复现。你可以：

    • 换入新的原始数据，不需要动模型代码
    • 扩展 engineer_features() 来添加新特征
    • 替换 pipeline 里的 estimator 来尝试不同模型
    • 为每个数据变换步骤编写单元测试
