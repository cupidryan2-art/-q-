    The Dataset: Student Performance Prediction

    We'll use a synthetic dataset of 400 students. Each row represents one student with four input features and one outcome. We'll predict both a continuous score (regression) and a pass/fail label (classification).

    ColumnTypeDescription
    study_hoursfloatWeekly study hours (0–20), with some outliers
    prev_scorefloatPrevious exam score (0–100), a few missing values
    attendancefloatAttendance rate (0.0–1.0)
    sleep_hoursfloatAverage nightly sleep (3–10)
    support_levelstrTutoring support: "none", "basic", "intensive"
    final_scorefloatTarget: final exam score (0–100)

    Step 0 — Generate the Synthetic Dataset

    We generate the data with NumPy so this project is fully self-contained — no external file needed. The data has realistic noise, missing values, and outliers baked in.

    Python — generate_data.py (run once)
    import numpy as np
import pandas as pd

rng = np.random.default_rng(seed=2024)
n = 400

study_hours   = rng.uniform(0, 16, n)
prev_score    = rng.normal(65, 18, n).clip(0, 100)
attendance    = rng.beta(7, 2, n)           # skewed toward high attendance
sleep_hours   = rng.normal(7, 1.2, n).clip(3, 10)
support_level = rng.choice(["none", "basic", "intensive"],
                             p=[0.5, 0.3, 0.2], size=n)

# True relationship (with noise)
support_bonus = {"none": 0, "basic": 5, "intensive": 12}
bonus = np.array([support_bonus[s] for s in support_level])

final_score = (
    0.35 * study_hours * 4           # study hours → up to ~22 pts
  + 0.40 * prev_score                # previous score carries weight
  + 0.15 * attendance * 100          # attendance → up to 15 pts
  + 0.10 * (sleep_hours - 4) * 5    # sleep → small bonus
  + bonus
  + rng.normal(0, 8, n)             # noise
).clip(0, 100)

df = pd.DataFrame({
    "study_hours":   study_hours,
    "prev_score":    prev_score,
    "attendance":    attendance,
    "sleep_hours":   sleep_hours,
    "support_level": support_level,
    "final_score":   final_score,
})

# Inject realistic problems
df.loc[rng.choice(n, 20, replace=False), "prev_score"] = np.nan   # missing values
df.loc[rng.choice(n, 5,  replace=False), "study_hours"] *= 3      # outliers
df.loc[rng.choice(n, 8,  replace=False), "attendance"] = rng.uniform(0, 0.1, 8)

df = df.round(2)
df.to_csv("students.csv", index=False)
print(f"Saved {len(df)} rows to students.csv")

    Step 2 — Clean the Data

    Python — 02_clean.py
    import pandas as pd
import numpy as np

df = pd.read_csv("students.csv")

def clean(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()

    # ── Missing values ──
    # Fill prev_score with median (robust to outliers)
    df["prev_score"] = df["prev_score"].fillna(df["prev_score"].median())

    # ── Outliers in study_hours (cap at 20h/week) ──
    df["study_hours"] = df["study_hours"].clip(upper=20.0)

    # ── Impossible attendance values ──
    df["attendance"] = df["attendance"].clip(0.0, 1.0)

    # ── Ensure dtypes ──
    df["support_level"] = df["support_level"].str.strip().str.lower()

    return df

df_clean = clean(df)

# Verify
print("Missing after cleaning:", df_clean.isnull().sum().sum())
print("study_hours max:", df_clean["study_hours"].max())
print("attendance range:", df_clean["attendance"].min(), "–", df_clean["attendance"].max())

df_clean.to_csv("students_clean.csv", index=False)
print(f"\nCleaned data saved. Shape: {df_clean.shape}")

    Step 3 — Feature Engineering

    Python — 03_features.py
    import pandas as pd
import numpy as np

df = pd.read_csv("students_clean.csv")

def engineer_features(df: pd.DataFrame) -> pd.DataFrame:
    df = df.copy()

    # ── Encode ordinal categorical ──
    support_map = {"none": 0, "basic": 1, "intensive": 2}
    df["support_num"] = df["support_level"].map(support_map)

    # ── Interaction features ──
    # Study efficiency = study hours × attendance
    df["study_efficiency"] = df["study_hours"] * df["attendance"]

    # Sleep quality score — penalise too little AND too much
    df["sleep_quality"] = 1.0 - (np.abs(df["sleep_hours"] - 7.5) / 4.5).clip(0, 1)

    # ── Binary target for classification ──
    df["passed"] = (df["final_score"] >= 60).astype(int)

    return df

df_feat = engineer_features(df)

# Check what we have
print(df_feat[["study_efficiency", "sleep_quality", "support_num", "passed"]].describe().round(2))
df_feat.to_csv("students_features.csv", index=False)
print(f"\nFeature data saved. Columns: {list(df_feat.columns)}")

    Step 4 — Build Models

    Python — 04_model.py
    import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.preprocessing import StandardScaler
from sklearn.pipeline import Pipeline
from sklearn.linear_model import LinearRegression, LogisticRegression
from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import (
    r2_score, mean_absolute_error,
    accuracy_score, classification_report
)

df = pd.read_csv("students_features.csv")

FEATURES = ["study_hours", "prev_score", "attendance",
            "sleep_hours", "support_num", "study_efficiency", "sleep_quality"]

X = df[FEATURES]
y_reg  = df["final_score"]    # regression target (continuous)
y_clf  = df["passed"]          # classification target (binary)

X_train, X_test, yr_train, yr_test, yc_train, yc_test = train_test_split(
    X, y_reg, y_clf, test_size=0.2, random_state=42, stratify=y_clf
)

print(f"Train: {len(X_train)} | Test: {len(X_test)}")
print(f"Pass rate (train): {yc_train.mean():.1%}  (test): {yc_test.mean():.1%}\n")

# ────────────────────────────────
#  REGRESSION — predict final_score
# ────────────────────────────────
reg_pipe = Pipeline([
    ("scaler", StandardScaler()),
    ("model",  LinearRegression()),
])
reg_pipe.fit(X_train, yr_train)
yr_pred = reg_pipe.predict(X_test)

print("=== Linear Regression ===")
print(f"R²:  {r2_score(yr_test, yr_pred):.3f}")
print(f"MAE: {mean_absolute_error(yr_test, yr_pred):.2f} points\n")

# Cross-validated R²
cv_r2 = cross_val_score(reg_pipe, X, y_reg, cv=5, scoring="r2")
print(f"5-fold CV R²: {cv_r2.mean():.3f} ± {cv_r2.std():.3f}\n")

# ────────────────────────────────
#  CLASSIFICATION — predict passed
# ────────────────────────────────
log_pipe = Pipeline([
    ("scaler", StandardScaler()),
    ("model",  LogisticRegression(max_iter=1000)),
])
log_pipe.fit(X_train, yc_train)
yc_pred = log_pipe.predict(X_test)

print("=== Logistic Regression ===")
print(f"Accuracy: {accuracy_score(yc_test, yc_pred):.3f}")
print(classification_report(yc_test, yc_pred, target_names=["Failed", "Passed"]))

# Decision Tree for comparison
tree_pipe = Pipeline([
    ("model", DecisionTreeClassifier(max_depth=5, min_samples_leaf=8, random_state=42))
])
tree_pipe.fit(X_train, yc_train)
print("=== Decision Tree ===")
print(f"Accuracy: {tree_pipe.score(X_test, yc_test):.3f}")

    Interpreting the Results

    Once you have numbers, the job isn't done — you need to ask whether they make sense and what they imply.

    Reading the Coefficients
    Because we standardised the features before fitting, the linear regression coefficients are directly comparable. A larger absolute coefficient means a stronger influence on the predicted score. If prev_score has the highest coefficient, it means previous performance is the strongest predictor — which makes intuitive sense and gives us confidence the model learned something real rather than noise.
    Python — sanity-checking predictions
    # Predict for two hypothetical students and verify the model makes sense

    Pipeline Thinking (管道思维)

    Notice what we did: every step is a pure function (Chapter 4) that takes a DataFrame and returns a new DataFrame. The pipeline is just function composition:

    Python — the complete pipeline in one function
    def run_pipeline(raw_csv: str) -> dict:
    """
    Complete data science pipeline — load to evaluation in one call.
    Each step is independently testable and replaceable.
    """
    # Load
    df = pd.read_csv(raw_csv)

    # Clean
    df = clean(df)                # Step 2

    # Feature engineering
    df = engineer_features(df)    # Step 3

    # Train / test split
    X = df[FEATURES]
    y_reg = df["final_score"]
    y_clf = df["passed"]
    X_train, X_test, yr_train, yr_test, yc_train, yc_test = train_test_split(
        X, y_reg, y_clf, test_size=0.2, random_state=42, stratify=y_clf
    )

    # Train models
    reg_pipe = Pipeline([("scaler", StandardScaler()), ("model", LinearRegression())])
    clf_pipe = Pipeline([("scaler", StandardScaler()), ("model", LogisticRegression(max_iter=1000))])
    reg_pipe.fit(X_train, yr_train)
    clf_pipe.fit(X_train, yc_train)

    # Evaluate
    return {
        "regression_r2":       r2_score(yr_test, reg_pipe.predict(X_test)),
        "regression_mae":      mean_absolute_error(yr_test, reg_pipe.predict(X_test)),
        "classification_acc":  clf_pipe.score(X_test, yc_test),
        "n_train": len(X_train),
        "n_test":  len(X_test),
    }

results = run_pipeline("students.csv")
for k, v in results.items():
    print(f"{k:25s}: {v:.3f}" if isinstance(v, float) else f"{k:25s}: {v}")
    What You've Built

    This is a real, production-ready data science workflow structure. The functions clean(), engineer_features(), and the final pipeline are independently testable, version-controllable, and reproducible. You can:

    • Swap in new raw data without touching the model code
    • Add new features by extending engineer_features()
    • Try different models by replacing the pipeline estimator
    • Write unit tests for each transformation step

    This is exactly how production ML systems are structured — and you now have the Python foundation to build them.
